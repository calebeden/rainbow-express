Create Database:
CREATE DATABASE `rainbow_express` DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci

Table for login information:
CREATE TABLE `users` (
	`id` int NOT NULL AUTO_INCREMENT,
	`name` varchar(255) DEFAULT NULL,
	`username` varchar(30) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
	`password` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
	`permissions` int NOT NULL DEFAULT '0',
	`properties` json DEFAULT NULL,
	PRIMARY KEY (`id`),
	UNIQUE KEY `username` (`username`)
) ENGINE=InnoDB AUTO_INCREMENT=0 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci

Table for storing apartment church data:
CREATE TABLE `properties` (
	`id` int NOT NULL AUTO_INCREMENT,
	`name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL,
	`street` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL,
	`city` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL,
	`state` char(2) COLLATE utf8mb4_unicode_ci NOT NULL,
	`zip` int NOT NULL,
	PRIMARY KEY (`id`)
) ENGINE=MyISAM AUTO_INCREMENT=0 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci

Table for storing when R.E. occurred
CREATE TABLE `dates` (
	`id` int NOT NULL AUTO_INCREMENT,
	`year` smallint NOT NULL,
 	`month` int NOT NULL,
	`date` int NOT NULL,
	`properties` json NOT NULL,
	PRIMARY KEY (`id`)
) ENGINE=MyISAM AUTO_INCREMENT=5 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
